breaking down a textual paragraph into small chunks such as words or
sentences. It is classified into two sections:
1. Sentence Tokenization and Word Tokenization:

5. Stemming: Stemming is a process of linguistics normalization to reduce
words to their word root or divide the derivational affixes.
For example: writing, wrote, written can stemmed or reduced as write.
import nltk
from nltk.stem import PorterStemmer
from nltk.tokenize import word_tokenize
# Create stemmer object
stemmer = PorterStemmer()
# Sample text
text = "running runs runner easily fairly cats studies studying"
# Tokenize text
words = word_tokenize(text)
# Apply stemming
stemmed_words = [stemmer.stem(word) for word in words]
print("Original Words:", words)
print("Stemmed Words:", stemmed_words)

# Remove special characters and digits
clean_text = re.sub(r'[^a-zA-Z\s]', '', text)
print(clean_text)
#Tokenize into sentences
sentences = sent_tokenize(clean_text)
print(sentences)
# Word tokenization and stopword removal
stop_words = set(stopwords.words('english'))
word_frequencies = {}
for word in word_tokenize(clean_text.lower()):
if word not in stop_words:
if word not in word_frequencies:
word_frequencies[word] = 1
else:
word_frequencies[word] += 1
# Normalize word frequencies
max_frequency = max(word_frequencies.values())
for word in word_frequencies:
word_frequencies[word] /= max_frequency
# Score sentences
sentence_scores = {}
for sentence in sentences:
for word in word_tokenize(sentence.lower()):
if word in word_frequencies:
if sentence not in sentence_scores:
